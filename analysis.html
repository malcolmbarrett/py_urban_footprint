import seaborn as sns
plot_data = (
    farms
        .filter(
            pl.col("SolarPanels_bin") != ""
        )
        .with_columns(
            pl.col("SizeOfProductionSite").log(),
            pl.col("WaterUse_GWP_2019").log(),
        )
)

sns.lmplot(
    data = plot_data,
    x="SizeOfProductionSite",
    y="WaterUse_GWP_2019",
    hue="FarmType",
    col="FarmType",
    palette="colorblind",
    legend="full"
)
